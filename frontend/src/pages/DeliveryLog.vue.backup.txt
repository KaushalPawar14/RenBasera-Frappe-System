<script setup>
import { ref, computed, onMounted, onUnmounted } from "vue";
import { useRouter } from "vue-router";
import { apiGet, apiPost, listUrl } from "../utils/api.js";
import { useAuthStore } from "../composables/useAuth";

const router = useRouter();
const auth = useAuthStore();
const loading = ref(true);
const showForm = ref(false);
const saving = ref(false);
const saveError = ref("");
const saveSuccess = ref("");
const activeTab = ref("pending");
const pendingChallans = ref([]);
const completedLogs = ref([]);
const selectedChallan = ref({});
const deliveryForm = ref([]);
const isOffline = ref(!navigator.onLine);
const pendingSync = ref([]);

// --- Offline detection ---
function handleOnline() {
	isOffline.value = false;
	loadPendingSync();
}
function handleOffline() {
	isOffline.value = true;
}

// --- Offline storage helpers ---
const OFFLINE_KEY = "tsf_delivery_offline";

function loadPendingSync() {
	try {
		const raw = localStorage.getItem(OFFLINE_KEY);
		pendingSync.value = raw ? JSON.parse(raw) : [];
	} catch {
		pendingSync.value = [];
	}
}

function saveToOffline(payload) {
	try {
		const existing = JSON.parse(localStorage.getItem(OFFLINE_KEY) || "[]");
		existing.push({ ...payload, _savedAt: new Date().toISOString() });
		localStorage.setItem(OFFLINE_KEY, JSON.stringify(existing));
		pendingSync.value = existing;
	} catch (e) {
		console.error("Offline save failed", e);
	}
}

function clearOfflineItem(index) {
	try {
		const existing = JSON.parse(localStorage.getItem(OFFLINE_KEY) || "[]");
		existing.splice(index, 1);
		localStorage.setItem(OFFLINE_KEY, JSON.stringify(existing));
		pendingSync.value = existing;
	} catch (e) {
		console.error(e);
	}
}

async function syncOfflineData() {
	if (isOffline.value || pendingSync.value.length === 0) return;
	let synced = 0;
	const toSync = [...pendingSync.value];

	for (let i = 0; i < toSync.length; i++) {
		const item = toSync[i];
		try {
			const res = await apiPost("/api/resource/TSF Delivery Log", item);
			if (res.data) {
				clearOfflineItem(0);
				synced++;
			}
		} catch (e) {
			console.error("Sync failed for item", i, e);
		}
	}

	if (synced > 0) {
		saveSuccess.value = `${synced} offline log(s) synced successfully!`;
		await fetchData();
		setTimeout(() => (saveSuccess.value = ""), 3000);
	}
}

// --- Helpers ---
function formatDate(d) {
	if (!d) return "";
	return new Date(d).toLocaleDateString("en-IN", {
		day: "2-digit",
		month: "short",
		year: "numeric",
	});
}

function getCenterCount(challan) {
	return challan.center_list?.length || 0;
}

const deliveredCount = computed(() => deliveryForm.value.filter((c) => c.delivered).length);

const progressPercent = computed(() =>
	deliveryForm.value.length > 0 ? (deliveredCount.value / deliveryForm.value.length) * 100 : 0,
);

function checkDelivered(center) {
	center.delivered = !!(center.arrival_time && center.delivered_qty > 0);
}

// --- Open Form ---
async function openForm(challan) {
	selectedChallan.value = challan;
	saveError.value = "";
	saveSuccess.value = "";

	// Load full challan to get center_list
	try {
		const res = await apiGet(`/api/resource/TSF Dispatch Challan/${challan.name}`);
		const full = res.data || {};
		const centers = full.center_list || [];
		const now = new Date();
		const localNow = new Date(now.getTime() - now.getTimezoneOffset() * 60000)
			.toISOString()
			.slice(0, 16);

		deliveryForm.value = centers.map((c) => ({
			center: c.name1 || c.center_name || c.name || "—",
			arrival_time: localNow,
			delivered_qty: c.delivered_qty || 0,
			delivered: false,
		}));

		// Mark as delivered if qty > 0
		deliveryForm.value.forEach((c) => checkDelivered(c));
	} catch (e) {
		deliveryForm.value = [];
		console.error(e);
	}

	showForm.value = true;
}

// --- Fetch Data ---
async function fetchData() {
	loading.value = true;
	try {
		// Get current user's driver record
		const driverRes = await apiGet(
			listUrl("TSF Driver", ["name", "driver_name", "user"], [["user", "=", auth.user]]),
		);
		const driverName = driverRes.data?.[0]?.name || null;

		// Get dispatched challans for this driver
		const filters = driverName
			? [
					["workflow_state", "=", "Dispatched"],
					["driver", "=", driverName],
				]
			: [["workflow_state", "=", "Dispatched"]];

		const [challanRes, logRes] = await Promise.all([
			apiGet(
				listUrl(
					"TSF Dispatch Challan",
					["name", "vehicle", "driver", "workflow_state", "creation", "center_list"],
					filters,
				),
			),
			apiGet(listUrl("TSF Delivery Log", ["name", "challan_ref", "creation"], [], 50)),
		]);

		pendingChallans.value = challanRes.data || [];
		completedLogs.value = logRes.data || [];

		// Add any offline saved logs to completed tab
		const offline = JSON.parse(localStorage.getItem(OFFLINE_KEY) || "[]");
		offline.forEach((o) => {
			completedLogs.value.unshift({
				name: "Offline",
				challan_ref: o.challan_ref,
				creation: o._savedAt,
				_offline: true,
			});
		});
	} catch (e) {
		console.error(e);
	}
	loading.value = false;
}

// --- Save ---
async function saveDeliveryLog() {
	if (deliveredCount.value === 0) {
		saveError.value = "Please fill at least one center delivery.";
		return;
	}
	saving.value = true;
	saveError.value = "";
	saveSuccess.value = "";

	const payload = {
		doctype: "TSF Delivery Log",
		challan_ref: selectedChallan.value.name,
		delivery_info: deliveryForm.value.map((c) => ({
			doctype: "TSF Delivery Log Item Table",
			center: c.center,
			arrival_time: c.arrival_time,
			delivered_qty: parseInt(c.delivered_qty) || 0,
		})),
	};

	if (isOffline.value) {
		saveToOffline(payload);
		saveSuccess.value = "Saved offline! Will sync when back online.";
		setTimeout(() => {
			showForm.value = false;
			fetchData();
		}, 1800);
		saving.value = false;
		return;
	}

	try {
		const res = await apiPost("/api/resource/TSF Delivery Log", payload);
		if (res.data) {
			saveSuccess.value = "Delivery log saved successfully!";
			setTimeout(() => {
				showForm.value = false;
				fetchData();
			}, 1800);
		} else {
			saveError.value = res.exception || res._server_messages || "Failed to save.";
			console.error(res);
		}
	} catch (e) {
		// Network error - save offline
		saveToOffline(payload);
		saveSuccess.value = "Network error — saved offline. Will sync later.";
		setTimeout(() => {
			showForm.value = false;
			fetchData();
		}, 1800);
	}
	saving.value = false;
}

onMounted(() => {
	window.addEventListener("online", handleOnline);
	window.addEventListener("offline", handleOffline);
	loadPendingSync();
	fetchData();
});

onUnmounted(() => {
	window.removeEventListener("online", handleOnline);
	window.removeEventListener("offline", handleOffline);
});
</script>
